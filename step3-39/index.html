这个插件依赖于 jQuery 开发 使用了 RequireJS 工具 使用方法： 

            {
                // 表格的数据
                data: {
                    // 表头的数据
                    head: ['姓名', '语文', '数学', '英语', '总分'],
                    // 表格内容的数据
                    body: [
                        ['小明', '80', '90', '70', '240'],
                        ['小红', '90', '60', '90', '240'],
                        ['小亮', '60', '100', '70', '230']
                    ]
                },
                tableClass: 'sort-table', // 给表格添加的类名
                sortClass: 'sort' // 给具有点击排序功能的表头单元格添加的类名
                fixHeader: false // 设置是否固定表头
            }
